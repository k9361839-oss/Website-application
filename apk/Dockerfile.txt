# שלב 1: שימוש בבסיס פייתון
FROM python:3.9-slim

# שלב 2: התקנת Java (דרוש עבור JADX) וכלים להורדה
RUN apt-get update && apt-get install -y \
    openjdk-17-jre-headless \
    wget \
    unzip \
    && rm -rf /var/lib/apt/lists/*

# שלב 3: הורדה והתקנה של JADX (גרסה יציבה)
RUN wget https://github.com/skylot/jadx/releases/download/v1.4.7/jadx-1.4.7.zip \
    && unzip jadx-1.4.7.zip -d /opt/jadx \
    && ln -s /opt/jadx/bin/jadx /usr/local/bin/jadx

# שלב 4: הגדרת תיקיית עבודה והעתקת קוד האתר
WORKDIR /app
COPY . /app

# שלב 5: התקנת הספריות של פייתון
RUN pip install --no-cache-dir flask

# שלב 6: הרצת האתר
CMD ["python", "app.py"]
